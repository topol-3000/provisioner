RESEARCH Pattern 1 (main.py TaskGroup + SIGTERM):
  shutdown = asyncio.Event()
  loop.add_signal_handler(signal.SIGTERM, shutdown.set)
  loop.add_signal_handler(signal.SIGINT, shutdown.set)
  async with asyncio.TaskGroup() as tg:
      tg.create_task(_run_consumer(settings, shutdown), name="consumer")
      tg.create_task(_run_convergence(settings, shutdown), name="convergence")
      tg.create_task(run_outbox_relay(settings, shutdown), name="outbox_relay")
      tg.create_task(run_health_server(settings, shutdown), name="health_server")
  Use "except* Exception as eg" (not "except Exception") — TaskGroup raises ExceptionGroup.

RESEARCH Pattern 2 (health server — no analog in platform-api):
  app = web.Application(); app.router.add_get("/healthz", _healthz)
  runner = web.AppRunner(app); await runner.setup()
  site = web.TCPSite(runner, "0.0.0.0", settings.health_port); await site.start()
  log.info("health server listening", port=settings.health_port)
  await shutdown.wait(); await runner.cleanup()
  _healthz returns web.Response(content_type="application/json", text='{"status":"ok"}')
